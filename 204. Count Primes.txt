
class Solution {
public:
    int countPrimes(long long n) {
        vector<bool>prime(n+1,1);
        prime[0]=prime[1]=false;
        int count=0;
        for(int i=2;i<n;i++)
        {
            if(prime[i])
            {
                count++;
                for(int j=2*i;j<n;j=j+i)
                {
                    prime[j]=false;
                }
            }
        }
        return count;
    }
};
